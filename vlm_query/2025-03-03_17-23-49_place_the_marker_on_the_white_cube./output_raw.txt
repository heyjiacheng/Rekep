# The task involves placing the marker on the white cube. This can be broken down into two stages:
# 1. Grasp the marker.
# 2. Place the marker on top of the white cube.

num_stages = 2

### stage 1 sub-goal constraints
def stage1_subgoal_constraint1(end_effector, keypoints):
    """Align the end-effector with the marker (keypoint 0)."""
    marker_position = keypoints[0]
    cost = np.linalg.norm(end_effector - marker_position)
    return cost

### stage 1 path constraints
# No path constraints needed for grasping.

### stage 2 sub-goal constraints

    offsetted_point = cube_position + np.array([0, 0, 0.1])  # Assume 10cm above the cube
    cost = np.linalg.norm(marker_position - offsetted_point)
    return cost
