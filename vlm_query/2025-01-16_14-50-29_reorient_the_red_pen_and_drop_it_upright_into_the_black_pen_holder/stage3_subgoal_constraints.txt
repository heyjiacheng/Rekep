def stage3_subgoal_constraint1(end_effector, keypoints):
    """The red pen (keypoint 1) is 10cm above the pen holder (keypoint 4)."""
    offsetted_point = keypoints[4] + np.array([0, 0, 0.1])
    return np.linalg.norm(keypoints[1] - offsetted_point)

def stage3_subgoal_constraint2(end_effector, keypoints):
    """The red pen (keypoint 1 to 2) must still stay upright (parallel to the z-axis)."""
    pen_vector = keypoints[2] - keypoints[1]
    z_axis = np.array([0, 0, 1])
    return np.linalg.norm(np.cross(pen_vector, z_axis))
